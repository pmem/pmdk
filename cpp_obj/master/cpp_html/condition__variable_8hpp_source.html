514  template <typename Clock, typename Duration>
515  std::cv_status
516  wait_until_impl(
517  mutex &lock,
518  const std::chrono::time_point<Clock, Duration> &abs_timeout)
519  {
520  PMEMobjpool *pop = pmemobj_pool_by_ptr(this);
521 
522  /* convert to my clock */
523  const typename Clock::time_point their_now = Clock::now();
524  const clock_type::time_point my_now = clock_type::now();
525  const auto delta = abs_timeout - their_now;
526  const auto my_rel = my_now + delta;
527 
528  struct timespec ts = detail::timepoint_to_timespec(my_rel);
529 
530  auto ret = pmemobj_cond_timedwait(pop, &this->pcond,
531  lock.native_handle(), &ts);
532 
533  if (ret == 0)
534  return std::cv_status::no_timeout;
535  else if (ret == ETIMEDOUT)
536  return std::cv_status::timeout;
537  else
538  throw lock_error(ret, std::system_category(),
539  "Error waiting on a condition "
540  "variable.");
541  }
