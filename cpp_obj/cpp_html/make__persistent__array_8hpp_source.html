65  template<typename T>
66  typename detail::pp_if_array<T>::type make_persistent(std::size_t N)
67  {
68  typedef typename detail::pp_array_type<T>::type I;
69 
70  if (pmemobj_tx_stage() != TX_STAGE_WORK)
71  throw transaction_scope_error("refusing to allocate "
72  "memory outside of transaction scope");
73 
74  persistent_ptr<T> ptr = pmemobj_tx_alloc(sizeof (I) * N,
75  detail::type_num<I>());
76 
77  if (ptr == nullptr)
78  throw transaction_alloc_error("failed to allocate "
79  "persistent memory array");
80 
81  std::size_t i;
82  try {
83  for (i = 0; i < N; ++i)
84  ::new (ptr.get() + i) I();
85  } catch (...) {
86  for (std::size_t j = 1; j <= i; ++j)
87  detail::destroy<I>(ptr[i - j]);
88  pmemobj_tx_free(*ptr.raw_ptr());
89  throw;
90  }
91 
92  return ptr;
93  }

155  template<typename T>
156  void delete_persistent(typename detail::pp_if_array<T>::type &ptr,
157  std::size_t N)
158  {
159  typedef typename detail::pp_array_type<T>::type I;
160 
161  if (pmemobj_tx_stage() != TX_STAGE_WORK)
162  throw transaction_scope_error("refusing to allocate "
163  "memory outside of transaction scope");
164 
165  if (ptr == nullptr)
166  return;
167 
168  for (std::size_t i = 0; i < N; ++i)
169  detail::destroy<I>(ptr[N - 1 - i]);
170 
171  if (pmemobj_tx_free(*ptr.raw_ptr()) != 0)
172  throw transaction_alloc_error("failed to delete "
173  "persistent memory object");
174 
175  /* to be able to revert the state of the persistent_ptr */
176  detail::conditional_add_to_tx(&ptr);
177  ptr = OID_NULL;
178  }
