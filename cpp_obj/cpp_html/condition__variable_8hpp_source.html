503  template<typename Clock, typename Duration>
504  std::cv_status wait_until_impl(mutex &lock,
505  const std::chrono::time_point<Clock,
506  Duration> &abs_timeout)
507  {
508  PMEMobjpool *pop = pmemobj_pool_by_ptr(this);
509 
510  /* convert to my clock */
511  const typename Clock::time_point their_now =
512  Clock::now();
513  const clock_type::time_point my_now = clock_type::now();
514  const auto delta = abs_timeout - their_now;
515  const auto my_rel = my_now + delta;
516 
517  struct timespec ts =
518  detail::timepoint_to_timespec(my_rel);
519 
520  auto ret = pmemobj_cond_timedwait(pop, &this->pcond,
521  lock.native_handle(), &ts);
522 
523  if (ret == 0)
524  return std::cv_status::no_timeout;
525  else if (ret == ETIMEDOUT)
526  return std::cv_status::timeout;
527  else
528  throw lock_error(ret, std::system_category(),
529  "Error waiting on a condition "
530  "variable.");
531  }

537  bool wait_until_impl(mutex &lock,
538  const std::chrono::time_point<Clock,
539  Duration> &abs_timeout, Predicate pred)
540  {
541  while (!pred())
542  if (this->wait_until_impl(lock, abs_timeout) ==
543  std::cv_status::timeout)
544  return pred();
545  return true;
